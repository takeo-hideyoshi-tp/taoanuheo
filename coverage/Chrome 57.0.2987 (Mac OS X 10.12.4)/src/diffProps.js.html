  function operateAttribute(dom, name, value, isSVG) {
 
      var method = value === '' ? 'removeAttribute' : 'setAttribute',
          namespace = null
      //http://www.w3school.com.cn/xlink/xlink_reference.asp

      if (isSVG && name.indexOf('xlink') === 0) {
